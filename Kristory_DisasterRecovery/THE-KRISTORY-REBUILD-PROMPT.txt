-- JOURNAL ENTRIES
CREATE TABLE journal_entries (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  entry_date DATE UNIQUE NOT NULL,
  created_at TIMESTAMPTZ DEFAULT now(),
  updated_at TIMESTAMPTZ DEFAULT now()
);
ALTER TABLE journal_entries DISABLE ROW LEVEL SECURITY;

-- ENTRY SECTIONS (one per user per day)
CREATE TABLE entry_sections (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  entry_id UUID REFERENCES journal_entries(id) ON DELETE CASCADE,
  user_id UUID REFERENCES users(id),
  content TEXT,
  created_at TIMESTAMPTZ DEFAULT now(),
  updated_at TIMESTAMPTZ DEFAULT now(),
  UNIQUE (entry_id, user_id)
);
ALTER TABLE entry_sections DISABLE ROW LEVEL SECURITY;

ALTER TABLE entry_sections ADD COLUMN fts TSVECTOR
  GENERATED ALWAYS AS (to_tsvector('english', coalesce(content, ''))) STORED;
CREATE INDEX idx_entry_sections_fts ON entry_sections USING GIN (fts);

-- ENTRY PHOTOS
CREATE TABLE entry_photos (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  entry_id UUID REFERENCES journal_entries(id) ON DELETE CASCADE,
  user_id UUID REFERENCES users(id),
  storage_path TEXT NOT NULL,
  display_order INTEGER DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT now()
);
ALTER TABLE entry_photos DISABLE ROW LEVEL SECURITY;

## CRITICAL TECHNICAL RULES

1. **FK Hints on ALL Supabase nested queries.** Every .select() that joins related tables MUST use !foreign_key syntax. Example:
   .select('*, sections:entry_sections!entry_id(*), photos:entry_photos!entry_id(*), tagged_items:tagged_items!entry_id(*, category:categories!category_id(*))')
   Without hints, PostgREST returns 400 errors. This is the #1 source of bugs.

2. **Optimistic UI everywhere.** Mutations update local cache immediately, sync in background. Show "Saving..." / "Saved ✓".

3. **Mobile-first.** 375px width first. 44px minimum tap targets. Bottom nav with safe area padding.

4. **Photo resize before upload:** max 1600px longest side, JPEG 85% quality, canvas-based. Use Math.round() on dimensions. Revoke object URLs after use.

5. **Auto-save with 1.5s debounce** on journal entry editing.

6. **PWA: skipWaiting + clientsClaim** for immediate service worker updates.
